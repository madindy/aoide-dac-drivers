For system that can use rpi-update.
curl -L --output ./dac_install.sh https://github.com/howardqiao/aoide-dac-drivers/raw/master/dac_install.sh && chmod +x ./dac_install.sh
sudo ./dac_install.sh

sudo ./aoidetft_setup.sh
